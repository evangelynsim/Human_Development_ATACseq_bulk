#!/bin/bash

set -x


#Create sets of foreground regions in bed format
for XLS in *xls ; do

  UP=$XLS.up.bed
  DN=$XLS.dn.bed

  awk ' $5<0.01 && $2>0 {print $1"\t"$1}' $XLS \
  | cut -d '_' -f2- | sed 's/:/\t/' | sed 's/-/\t/' \
  | bedtools sort > $UP

  awk ' $5<0.01 && $2<0 {print $1"\t"$1}' $XLS \
  | cut -d '_' -f2- | sed 's/:/\t/' | sed 's/-/\t/' \
  | bedtools sort > $DN
done

#Create a set of background regions from those not changing in any comparison
BG=$XLS.bg.bed
tail -qn +2 *xls \
| awk '$5>0.1 {print $1}' \
| sort | uniq -c \
| cut -d '_' -f2- | tr ':-' '\t' \
| bedtools sort > $BG
